        public void CaricaMerende()
        {
            string line;
            StreamReader sr = new StreamReader("ListaMerende.csv");
            sr.ReadLine();
            while ((line = sr.ReadLine()) != null)
            {
                try
                {
                    //Creo una istanza 
                    Merenda m = new Merenda();
                    //Vettore
                    string[] campi = line.Split(';');
                    m.Nome = campi[0];
                    double Prezzo = Convert.ToDouble(campi[1]);
                    m.Prezzo = Prezzo;
                    Merende.Add(m);

                }
                catch (Exception e)
                {
                    MessageBox.Show(e.ToString());
                    break;
                }

                
            }

            foreach (Merenda m in Merende)
            {
                Lst_MerendeDisponibili.Items.Add(m);
            }
        }
